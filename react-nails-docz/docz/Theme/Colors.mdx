import { Playground } from "docz";
import styled from 'styled-components'
import { Row,Grid, ThemeProps, Filler, Column,Types, Text, Box, NailsBox } from "react-nails";
const {ItemAlignType} = Types
const ColorBox = styled(NailsBox).attrs(()=>({flex: true}))`
    min-height: 8em;
`
const ColorField = styled(ColorBox)`
    background-color: ${p=>p.backgroundColor};
`

# Colors

The listed colors can be used to set the `backgroundColor` property of elements.

<Grid columns={12} itemSpace={["small", "medium"]} lineSpace={["small", "medium"]}>
    <Grid.Row>
        <ThemeProps>
            {(theme) => (
                Object.entries(theme.colors)
                    .sort((a, b) => a[0].localeCompare(b[0]))
                    .map(([col, value]) =>
                        <Grid.Column width={[6, 3, 2]} key={col}>
                            <Box as={ColorBox} backgroundColor={col} space="medium" textColor={col === "text" ? "textInvert" : "text"} rounded>
                                <Filler stretchChild>
                                    <Column>
                                        <Text textWeight="bold">{col}</Text>
                                        <Filler align={ItemAlignType.Bottom} stretchChild>
                                            <Text textAlign="right" block>{value}</Text>
                                        </Filler>
                                    </Column>
                                </Filler>
                            </Box>
                        </Grid.Column>
                    )
            )}
        </ThemeProps>
    </Grid.Row>
</Grid>

# Palette

These colors are defined inside the theme and can be used as explicit property inside components.
<Grid columns={12} itemSpace={["small", "medium"]} lineSpace={["small", "medium"]}>
    <Grid.Row>
        <ThemeProps>
            {(theme) => (
                Object.entries(theme.palette)
                    .sort((a, b) => a[0].localeCompare(b[0]))
                    .map(([col, value]) =>
                        <Grid.Column width={[3, 2]} key={col}>
                            <Box as={ColorField} backgroundColor={value} space="medium" textColor={value === theme.colors.text ? "textInvert" : "text"} rounded>
                                <Filler stretchChild>
                                    <Column>
                                        <Text textWeight="bold">{col}</Text>
                                        <Filler align={ItemAlignType.Bottom} stretchChild>
                                            <Text textAlign="right" block>{value}</Text>
                                        </Filler>
                                    </Column>
                                </Filler>
                            </Box>
                        </Grid.Column>
                    )
            )}
        </ThemeProps>
    </Grid.Row>
</Grid>
